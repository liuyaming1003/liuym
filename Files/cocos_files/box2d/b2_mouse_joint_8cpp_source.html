00056 void b2MouseJoint::InitVelocityConstraints(const b2TimeStep& step)
00057 {
00058         b2Body* b = m_bodyB;
00059 
00060         float32 mass = b->GetMass();
00061 
00062         // Frequency
00063         float32 omega = 2.0f * b2_pi * m_frequencyHz;
00064 
00065         // Damping coefficient
00066         float32 d = 2.0f * mass * m_dampingRatio * omega;
00067 
00068         // Spring stiffness
00069         float32 k = mass * (omega * omega);
00070 
00071         // magic formulas
00072         // gamma has units of inverse mass.
00073         // beta has units of inverse time.
00074         b2Assert(d + step.dt * k > B2_FLT_EPSILON);
00075         m_gamma = step.dt * (d + step.dt * k);
00076         if (m_gamma != 0.0f)
00077         {
00078                 m_gamma = 1.0f / m_gamma;
00079         }
00080         m_beta = step.dt * k * m_gamma;
00081 
00082         // Compute the effective mass matrix.
00083         b2Vec2 r = b2Mul(b->GetTransform().R, m_localAnchor - b->GetLocalCenter());
00084 
00085         // K    = [(1/m1 + 1/m2) * eye(2) - skew(r1) * invI1 * skew(r1) - skew(r2) * invI2 * skew(r2)]
00086         //      = [1/m1+1/m2     0    ] + invI1 * [r1.y*r1.y -r1.x*r1.y] + invI2 * [r1.y*r1.y -r1.x*r1.y]
00087         //        [    0     1/m1+1/m2]           [-r1.x*r1.y r1.x*r1.x]           [-r1.x*r1.y r1.x*r1.x]
00088         float32 invMass = b->m_invMass;
00089         float32 invI = b->m_invI;
00090 
00091         b2Mat22 K1;
00092         K1.col1.x = invMass;    K1.col2.x = 0.0f;
00093         K1.col1.y = 0.0f;               K1.col2.y = invMass;
00094 
00095         b2Mat22 K2;
00096         K2.col1.x =  invI * r.y * r.y;  K2.col2.x = -invI * r.x * r.y;
00097         K2.col1.y = -invI * r.x * r.y;  K2.col2.y =  invI * r.x * r.x;
00098 
00099         b2Mat22 K = K1 + K2;
00100         K.col1.x += m_gamma;
00101         K.col2.y += m_gamma;
00102 
00103         m_mass = K.GetInverse();
00104 
00105         m_C = b->m_sweep.c + r - m_target;
00106 
00107         // Cheat with some damping
00108         b->m_angularVelocity *= 0.98f;
00109 
00110         // Warm starting.
00111         m_impulse *= step.dtRatio;
00112         b->m_linearVelocity += invMass * m_impulse;
00113         b->m_angularVelocity += invI * b2Cross(r, m_impulse);
00114 }

00116 void b2MouseJoint::SolveVelocityConstraints(const b2TimeStep& step)
00117 {
00118         b2Body* b = m_bodyB;
00119 
00120         b2Vec2 r = b2Mul(b->GetTransform().R, m_localAnchor - b->GetLocalCenter());
00121 
00122         // Cdot = v + cross(w, r)
00123         b2Vec2 Cdot = b->m_linearVelocity + b2Cross(b->m_angularVelocity, r);
00124         b2Vec2 impulse = b2Mul(m_mass, -(Cdot + m_beta * m_C + m_gamma * m_impulse));
00125 
00126         b2Vec2 oldImpulse = m_impulse;
00127         m_impulse += impulse;
00128         float32 maxImpulse = step.dt * m_maxForce;
00129         if (m_impulse.LengthSquared() > maxImpulse * maxImpulse)
00130         {
00131                 m_impulse *= maxImpulse / m_impulse.Length();
00132         }
00133         impulse = m_impulse - oldImpulse;
00134 
00135         b->m_linearVelocity += b->m_invMass * impulse;
00136         b->m_angularVelocity += b->m_invI * b2Cross(r, impulse);
00137 }
