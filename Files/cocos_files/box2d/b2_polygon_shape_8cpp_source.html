00094         // It's location doesn't change the result (except for rounding error).

00201 void b2PolygonShape::RayCast(b2RayCastOutput* output, const b2RayCastInput& input, const b2Transform& xf) const
00202 {
00203         float32 lower = 0.0f, upper = input.maxFraction;
00204 
00205         // Put the ray into the polygon's frame of reference.
00206         b2Vec2 p1 = b2MulT(xf.R, input.p1 - xf.position);
00207         b2Vec2 p2 = b2MulT(xf.R, input.p2 - xf.position);
00208         b2Vec2 d = p2 - p1;
00209         int32 index = -1;
00210 
00211         output->hit = false;
00212 
00213         for (int32 i = 0; i < m_vertexCount; ++i)
00214         {
00215                 // p = p1 + a * d
00216                 // dot(normal, p - v) = 0
00217                 // dot(normal, p1 - v) + a * dot(normal, d) = 0
00218                 float32 numerator = b2Dot(m_normals[i], m_vertices[i] - p1);
00219                 float32 denominator = b2Dot(m_normals[i], d);
00220 
00221                 if (denominator == 0.0f)
00222                 {       
00223                         if (numerator < 0.0f)
00224                         {
00225                                 return;
00226                         }
00227                 }
00228                 else
00229                 {
00230                         // Note: we want this predicate without division:
00231                         // lower < numerator / denominator, where denominator < 0
00232                         // Since denominator < 0, we have to flip the inequality:
00233                         // lower < numerator / denominator <==> denominator * lower > numerator.
00234                         if (denominator < 0.0f && numerator < lower * denominator)
00235                         {
00236                                 // Increase lower.
00237                                 // The segment enters this half-space.
00238                                 lower = numerator / denominator;
00239                                 index = i;
00240                         }
00241                         else if (denominator > 0.0f && numerator < upper * denominator)
00242                         {
00243                                 // Decrease upper.
00244                                 // The segment exits this half-space.
00245                                 upper = numerator / denominator;
00246                         }
00247                 }
00248 
00249                 if (upper < lower)
00250                 {
00251                         return;
00252                 }
00253         }
00254 
00255         b2Assert(0.0f <= lower && lower <= input.maxFraction);
00256 
00257         if (index >= 0)
00258         {
00259                 output->hit = true;
00260                 output->fraction = lower;
00261                 output->normal = b2Mul(xf.R, m_normals[index]);
00262                 return;

00283 void b2PolygonShape::ComputeMass(b2MassData* massData, float32 density) const
00284 {
00285         // Polygon mass, centroid, and inertia.
00286         // Let rho be the polygon density in mass per unit area.
00287         // Then:
00288         // mass = rho * int(dA)
00289         // centroid.x = (1/mass) * rho * int(x * dA)
00290         // centroid.y = (1/mass) * rho * int(y * dA)
00291         // I = rho * int((x*x + y*y) * dA)
00292         //
00293         // We can compute these integrals by summing all the integrals
00294         // for each triangle of the polygon. To evaluate the integral
00295         // for a single triangle, we make a change of variables to
00296         // the (u,v) coordinates of the triangle:
00297         // x = x0 + e1x * u + e2x * v
00298         // y = y0 + e1y * u + e2y * v
00299         // where 0 <= u && 0 <= v && u + v <= 1.
00300         //
00301         // We integrate u from [0,1-v] and then v from [0,1].
00302         // We also need to use the Jacobian of the transformation:
00303         // D = cross(e1, e2)
00304         //
00305         // Simplification: triangle centroid = (1/3) * (p1 + p2 + p3)
00306         //
00307         // The rest of the derivation is handled by computer algebra.
00308 
00309         b2Assert(m_vertexCount >= 2);
00310 
00311         // A line segment has zero mass.
00312         if (m_vertexCount == 2)
00313         {
00314                 massData->center = 0.5f * (m_vertices[0] + m_vertices[1]);
00315                 massData->mass = 0.0f;
00316                 massData->I = 0.0f;
00317                 return;
00318         }
00319 
00320         b2Vec2 center; center.Set(0.0f, 0.0f);
00321         float32 area = 0.0f;
00322         float32 I = 0.0f;
00323 
00324         // pRef is the reference point for forming triangles.
00325         // It's location doesn't change the result (except for rounding error).
00326         b2Vec2 pRef(0.0f, 0.0f);
00327 #if 0
00328         // This code would put the reference point inside the polygon.
00329         for (int32 i = 0; i < m_vertexCount; ++i)
00330         {
00331                 pRef += m_vertices[i];
00332         }
00333         pRef *= 1.0f / count;
00334 #endif
00335 
00336         const float32 k_inv3 = 1.0f / 3.0f;
00337 
00338         for (int32 i = 0; i < m_vertexCount; ++i)
00339         {
00340                 // Triangle vertices.
00341                 b2Vec2 p1 = pRef;
00342                 b2Vec2 p2 = m_vertices[i];
00343                 b2Vec2 p3 = i + 1 < m_vertexCount ? m_vertices[i+1] : m_vertices[0];
00344 
00345                 b2Vec2 e1 = p2 - p1;
00346                 b2Vec2 e2 = p3 - p1;
00347 
00348                 float32 D = b2Cross(e1, e2);
00349 
00350                 float32 triangleArea = 0.5f * D;
00351                 area += triangleArea;
00352 
00353                 // Area weighted centroid
00354                 center += triangleArea * k_inv3 * (p1 + p2 + p3);
00355 
00356                 float32 px = p1.x, py = p1.y;
00357                 float32 ex1 = e1.x, ey1 = e1.y;
00358                 float32 ex2 = e2.x, ey2 = e2.y;
00359 
00360                 float32 intx2 = k_inv3 * (0.25f * (ex1*ex1 + ex2*ex1 + ex2*ex2) + (px*ex1 + px*ex2)) + 0.5f*px*px;
00361                 float32 inty2 = k_inv3 * (0.25f * (ey1*ey1 + ey2*ey1 + ey2*ey2) + (py*ey1 + py*ey2)) + 0.5f*py*py;
00362 
00363                 I += D * (intx2 + inty2);
00364         }
00365 
00366         // Total mass
00367         massData->mass = density * area;
00368 
00369         // Center of mass
00370         b2Assert(area > B2_FLT_EPSILON);
00371         center *= 1.0f / area;
00372         massData->center = center;
00373 
00374         // Inertia tensor relative to the local origin.
00375         massData->I = density * I;
00376 }
