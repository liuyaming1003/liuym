00037 // Collision Detection in Interactive 3D Environments by Gino van den Bergen
00038 // From Section 3.1.2
00039 // x = s + a * r
00040 // norm(x) = radius
00041 void b2CircleShape::RayCast(b2RayCastOutput* output, const b2RayCastInput& input, const b2Transform& transform) const
00042 {
00043         b2Vec2 position = transform.position + b2Mul(transform.R, m_p);
00044         b2Vec2 s = input.p1 - position;
00045         float32 b = b2Dot(s, s) - m_radius * m_radius;
00046 
00047         // Solve quadratic equation.
00048         b2Vec2 r = input.p2 - input.p1;
00049         float32 c =  b2Dot(s, r);
00050         float32 rr = b2Dot(r, r);
00051         float32 sigma = c * c - rr * b;
00052 
00053         // Check for negative discriminant and short segment.
00054         if (sigma < 0.0f || rr < B2_FLT_EPSILON)
00055         {
00056                 output->hit = false;
00057                 return;
00058         }
00059 
00060         // Find the point of intersection of the line with the circle.
00061         float32 a = -(c + b2Sqrt(sigma));
00062 
00063         // Is the intersection point on the segment?
00064         if (0.0f <= a && a <= input.maxFraction * rr)
00065         {
00066                 a /= rr;
00067                 output->hit = true;
00068                 output->fraction = a;
00069                 output->normal = s + a * r;
00070                 output->normal.Normalize();
00071                 return;
00072         }
00073 
00074         output->hit = false;
00075         return;
00076 }
