00235                 for (int32 j = 0; j < c->pointCount; ++j)
00236                 {
00237                         b2ContactConstraintPoint* ccp = c->points + j;
00238 
00239                         // Relative velocity at contact
00240                         b2Vec2 dv = vB + b2Cross(wB, ccp->rB) - vA - b2Cross(wA, ccp->rA);
00241 
00242                         // Compute tangent force
00243                         float32 vt = b2Dot(dv, tangent);
00244                         float32 lambda = ccp->tangentMass * (-vt);
00245 
00246                         // b2Clamp the accumulated force
00247                         float32 maxFriction = friction * ccp->normalImpulse;
00248                         float32 newImpulse = b2Clamp(ccp->tangentImpulse + lambda, -maxFriction, maxFriction);
00249                         lambda = newImpulse - ccp->tangentImpulse;

00251                         // Apply contact impulse
00252                         b2Vec2 P = lambda * tangent;
00253 
00254                         vA -= invMassA * P;
00255                         wA -= invIA * b2Cross(ccp->rA, P);
00256 
00257                         vB += invMassB * P;
00258                         wB += invIB * b2Cross(ccp->rB, P);
00259 
00260                         ccp->tangentImpulse = newImpulse;
00261                 }
00262 
00263                 // Solve normal constraints
00264                 if (c->pointCount == 1)
00265                 {
00266                         b2ContactConstraintPoint* ccp = c->points + 0;
00267 
00268                         // Relative velocity at contact
00269                         b2Vec2 dv = vB + b2Cross(wB, ccp->rB) - vA - b2Cross(wA, ccp->rA);

00271                         // Compute normal impulse
00272                         float32 vn = b2Dot(dv, normal);
00273                         float32 lambda = -ccp->normalMass * (vn - ccp->velocityBias);

00275                         // b2Clamp the accumulated impulse
00276                         float32 newImpulse = b2Max(ccp->normalImpulse + lambda, 0.0f);
00277                         lambda = newImpulse - ccp->normalImpulse;

00279                         // Apply contact impulse
00280                         b2Vec2 P = lambda * normal;

00282                         wA -= invIA * b2Cross(ccp->rA, P);
00283 
00284                         vB += invMassB * P;
00285                         wB += invIB * b2Cross(ccp->rB, P);
00286                         ccp->normalImpulse = newImpulse;
00287                 }
00288                 else
00289                 {
00290                         // Block solver developed in collaboration with Dirk Gregorius (back in 01/07 on Box2D_Lite).
00291                         // Build the mini LCP for this contact patch
00292                         //
00293                         // vn = A * x + b, vn >= 0, , vn >= 0, x >= 0 and vn_i * x_i = 0 with i = 1..2
00294                         //
00295                         // A = J * W * JT and J = ( -n, -r1 x n, n, r2 x n )
00296                         // b = vn_0 - velocityBias
00297                         //
00298                         // The system is solved using the "Total enumeration method" (s. Murty). The complementary constraint vn_i * x_i
00299                         // implies that we must have in any solution either vn_i = 0 or x_i = 0. So for the 2D contact problem the cases
00300                         // vn1 = 0 and vn2 = 0, x1 = 0 and x2 = 0, x1 = 0 and vn2 = 0, x2 = 0 and vn1 = 0 need to be tested. The first valid
00301                         // solution that satisfies the problem is chosen.
00302                         // 
00303                         // In order to account of the accumulated impulse 'a' (because of the iterative nature of the solver which only requires
00304                         // that the accumulated impulse is clamped and not the incremental impulse) we change the impulse variable (x_i).
00305                         //
00306                         // Substitute:
00307                         // 
00308                         // x = x' - a
00309                         // 
00310                         // Plug into above equation:
00311                         //
00312                         // vn = A * x + b
00313                         //    = A * (x' - a) + b
00314                         //    = A * x' + b - A * a
00315                         //    = A * x' + b'
