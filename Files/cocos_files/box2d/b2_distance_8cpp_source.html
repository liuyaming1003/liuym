00240 // Solve a line segment using barycentric coordinates.
00241 //
00242 // p = a1 * w1 + a2 * w2
00243 // a1 + a2 = 1
00244 //
00245 // The vector from the origin to the closest point on the line is
00246 // perpendicular to the line.
00247 // e12 = w2 - w1
00248 // dot(p, e) = 0
00249 // a1 * dot(w1, e) + a2 * dot(w2, e) = 0
00250 //
00251 // 2-by-2 linear system
00252 // [1      1     ][a1] = [1]
00253 // [w1.e12 w2.e12][a2] = [0]
00254 //
00255 // Define
00256 // d12_1 =  dot(w2, e12)
00257 // d12_2 = -dot(w1, e12)
00258 // d12 = d12_1 + d12_2
00259 //
00260 // Solution
00261 // a1 = d12_1 / d12
00262 // a2 = d12_2 / d12
00263 void b2Simplex::Solve2()
00264 {
00265         b2Vec2 w1 = m_v1.w;
00266         b2Vec2 w2 = m_v2.w;
00267         b2Vec2 e12 = w2 - w1;
00268 
00269         // w1 region
00270         float32 d12_2 = -b2Dot(w1, e12);
00271         if (d12_2 <= 0.0f)
00272         {
00273                 // a2 <= 0, so we clamp it to 0
00274                 m_v1.a = 1.0f;
00275                 m_count = 1;
00276                 return;
00277         }
00278 
00279         // w2 region
00280         float32 d12_1 = b2Dot(w2, e12);
00281         if (d12_1 <= 0.0f)
00282         {
00283                 // a1 <= 0, so we clamp it to 0
00284                 m_v2.a = 1.0f;
00285                 m_count = 1;
00286                 m_v1 = m_v2;
00287                 return;
00288         }
00289 
00290         // Must be in e12 region.
00291         float32 inv_d12 = 1.0f / (d12_1 + d12_2);
00292         m_v1.a = d12_1 * inv_d12;
00293         m_v2.a = d12_2 * inv_d12;
00294         m_count = 2;
00295 }
