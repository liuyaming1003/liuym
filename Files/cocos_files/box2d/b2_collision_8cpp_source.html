00133 // Collision Detection in Interactive 3D Environments by Gino van den Bergen
00134 // From Section 3.4.1
00135 // x = mu1 * p1 + mu2 * p2
00136 // mu1 + mu2 = 1 && mu1 >= 0 && mu2 >= 0
00137 // mu1 = 1 - mu2;
00138 // x = (1 - mu2) * p1 + mu2 * p2
00139 //   = p1 + mu2 * (p2 - p1)
00140 // x = s + a * r (s := start, r := end - start)
00141 // s + a * r = p1 + mu2 * d (d := p2 - p1)
00142 // -a * r + mu2 * d = b (b := s - p1)
00143 // [-r d] * [a; mu2] = b
00144 // Cramer's rule:
00145 // denom = det[-r d]
00146 // a = det[b d] / denom
00147 // mu2 = det[-r b] / denom
00148 bool b2Segment::TestSegment(float32* lambda, b2Vec2* normal, const b2Segment& segment, float32 maxLambda) const
00149 {
00150         b2Vec2 s = segment.p1;
00151         b2Vec2 r = segment.p2 - s;
00152         b2Vec2 d = p2 - p1;
00153         b2Vec2 n = b2Cross(d, 1.0f);
00154 
00155         const float32 k_slop = 100.0f * B2_FLT_EPSILON;
00156         float32 denom = -b2Dot(r, n);
00157 
00158         // Cull back facing collision and ignore parallel segments.
00159         if (denom > k_slop)
00160         {
00161                 // Does the segment intersect the infinite line associated with this segment?
00162                 b2Vec2 b = s - p1;
00163                 float32 a = b2Dot(b, n);
00164 
00165                 if (0.0f <= a && a <= maxLambda * denom)
00166                 {
00167                         float32 mu2 = -r.x * b.y + r.y * b.x;
00168 
00169                         // Does the segment intersect this segment?
00170                         if (-k_slop * denom <= mu2 && mu2 <= denom * (1.0f + k_slop))
00171                         {
00172                                 a /= denom;
00173                                 n.Normalize();
00174                                 *lambda = a;
00175                                 *normal = n;
00176                                 return true;
00177                         }
00178                 }
00179         }
00180 
00181         return false;
