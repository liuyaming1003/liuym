00102 void b2Fixture::Synchronize(b2BroadPhase* broadPhase, const b2Transform& transform1, const b2Transform& transform2)
00103 {
00104         if (m_proxyId == b2BroadPhase::e_nullProxy)
00105         {       
00106                 return;
00107         }
00108 
00109         // Compute an AABB that covers the swept shape (may miss some rotation effect).
00110         b2AABB aabb1, aabb2;
00111         m_shape->ComputeAABB(&aabb1, transform1);
00112         m_shape->ComputeAABB(&aabb2, transform2);
00113         
00114         m_aabb.Combine(aabb1, aabb2);
00115 
00116         b2Vec2 displacement = transform2.position - transform1.position;
00117 
00118         broadPhase->MoveProxy(m_proxyId, m_aabb, displacement);
