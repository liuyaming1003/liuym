00202 void b2Body::ResetMass()
00203 {
00204         // Compute mass data from shapes. Each shape has its own density.
00205         m_mass = 0.0f;
00206         m_invMass = 0.0f;
00207         m_I = 0.0f;
00208         m_invI = 0.0f;
00209 
00210         b2Vec2 center = b2Vec2_zero;
00211         for (b2Fixture* f = m_fixtureList; f; f = f->m_next)
00212         {
00213                 const b2MassData& massData = f->GetMassData();
00214                 m_mass += massData.mass;
00215                 center += massData.mass * massData.center;
00216                 m_I += massData.I;
00217         }
00218 
00219         // Compute center of mass.
00220         if (m_mass > 0.0f)
00221         {
00222                 m_invMass = 1.0f / m_mass;
00223                 center *= m_invMass;
00224         }
00225 
00226         if (m_I > 0.0f && (m_flags & e_fixedRotationFlag) == 0)
00227         {
00228                 // Center the inertia about the center of mass.
00229                 m_I -= m_mass * b2Dot(center, center);
00230                 b2Assert(m_I > 0.0f);
00231                 m_invI = 1.0f / m_I;
00232         }
00233         else
00234         {
00235                 m_I = 0.0f;
00236                 m_invI = 0.0f;
00237         }
00238 
00239         // Move center of mass.
00240         b2Vec2 oldCenter = m_sweep.c;
00241         m_sweep.localCenter = center;
00242         m_sweep.c0 = m_sweep.c = b2Mul(m_xf, m_sweep.localCenter);
00243 
00244         // Update center of mass velocity.
00245         m_linearVelocity += b2Cross(m_angularVelocity, m_sweep.c - oldCenter);
00246 
00247         // Determine the new body type.

00249         if (m_invMass == 0.0f && m_invI == 0.0f)
00250         {
