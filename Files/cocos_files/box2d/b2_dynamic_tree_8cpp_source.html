00053 {

00064                 b2Free(oldNodes);

00067                 // pointer becomes the "next" pointer.
00068                 for (int32 i = m_nodeCount; i < m_nodeCapacity - 1; ++i)
00069                 {
00070                         m_nodes[i].next = i + 1;
00071                 }
00072                 m_nodes[m_nodeCapacity-1].next = b2_nullNode;
