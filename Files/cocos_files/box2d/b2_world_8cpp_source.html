00371         for (b2Body* seed = m_bodyList; seed; seed = seed->m_next)
00372         {
00373                 if (seed->m_flags & (b2Body::e_islandFlag | b2Body::e_sleepFlag))

00375                         continue;
00376                 }
00377 
00378                 if (seed->IsStatic())
00379                 {
00380                         continue;

00428                                 {
00429                                         continue;
00430                                 }

00436

00438                         for (b2JointEdge* je = b->m_jointList; je; je = je->next)
00439                         {

00690                                 if (island.m_contactCount == island.m_contactCapacity)
00691                                 {
00692                                         break;

00695                                 // Has this contact already been added to an island?

00697                                 {
00698                                         continue;
00699                                 }
00700

00702                                 if (cEdge->contact->IsSolid() == false || cEdge->contact->IsTouching() == false)
00703                                 {

00707                                 island.Add(cEdge->contact);

00710                                 // Update other body.
00711                                 b2Body* other = cEdge->other;
00712 
00713                                 // Was the other body already added to this island?
00714                                 if (other->m_flags & b2Body::e_islandFlag)
00715                                 {
00716                                         continue;
00717                                 }

00719                                 // March forward, this can do no harm since this is the min TOI.

00727                                 queue[queueStart + queueSize] = other;
00728                                 ++queueSize;

00730                         }
00731 
00732                         for (b2JointEdge* jEdge = b->m_jointList; jEdge; jEdge = jEdge->next)
00733                         {
00734                                 if (island.m_jointCount == island.m_jointCapacity)
00735                                 {
00736                                         continue;
00737                                 }
00738 
00739                                 if (jEdge->joint->m_islandFlag == true)
00740                                 {
00741                                         continue;
00742                                 }
00743 
00744                                 island.Add(jEdge->joint);
00745 
00746                                 jEdge->joint->m_islandFlag = true;
00747 
00748                                 b2Body* other = jEdge->other;
00749 
00750                                 if (other->m_flags & b2Body::e_islandFlag)
00751                                 {
00752                                         continue;
00753                                 }
00754 
00755                                 if (!other->IsStatic())
00756                                 {
00757                                         other->Advance(minTOI);
00758                                         other->WakeUp();
00759                                 }
00760 
00761                                 b2Assert(queueStart + queueSize < queueCapacity);
00762                                 queue[queueStart + queueSize] = other;
00763                                 ++queueSize;
00764                                 other->m_flags |= b2Body::e_islandFlag;
00765                         }
00766                 }
00767 
00768                 b2TimeStep subStep;
00769                 subStep.warmStarting = false;
00770                 subStep.dt = (1.0f - minTOI) * step.dt;
00771                 subStep.inv_dt = 1.0f / subStep.dt;
00772                 subStep.dtRatio = 0.0f;
00773                 subStep.velocityIterations = step.velocityIterations;
00774                 subStep.positionIterations = step.positionIterations;
00775 
00776                 island.SolveTOI(subStep);
00777 
00778                 // Post solve cleanup.
00779                 for (int32 i = 0; i < island.m_bodyCount; ++i)
00780                 {
00781                         // Allow bodies to participate in future TOI islands.
00782                         b2Body* b = island.m_bodies[i];
00783                         b->m_flags &= ~b2Body::e_islandFlag;
00784 
00785                         if (b->m_flags & b2Body::e_sleepFlag)
00786                         {
00787                                 continue;
00788                         }
00789 
00790                         if (b->IsStatic())
00791                         {
00792                                 continue;
00793                         }
00794 
00795                         // Update fixtures (for broad-phase).
00796                         b->SynchronizeFixtures();
00797 
00798                         // Invalidate all contact TOIs associated with this body. Some of these
00799                         // may not be in the island because they were not touching.
00800                         for (b2ContactEdge* ce = b->m_contactList; ce; ce = ce->next)
00801                         {
00802                                 ce->contact->m_flags &= ~b2Contact::e_toiFlag;
00803                         }
00804                 }
00805 
00806                 for (int32 i = 0; i < island.m_contactCount; ++i)
00807                 {
00808                         // Allow contacts to participate in future TOI islands.
00809                         b2Contact* c = island.m_contacts[i];
00810                         c->m_flags &= ~(b2Contact::e_toiFlag | b2Contact::e_islandFlag);
00811                 }
00812 
00813                 for (int32 i = 0; i < island.m_jointCount; ++i)
00814                 {
00815                         // Allow joints to participate in future TOI islands.
00816                         b2Joint* j = island.m_joints[i];
00817                         j->m_islandFlag = false;
00818                 }
00819 
00820                 // Commit fixture proxy movements to the broad-phase so that new contacts are created.
00821                 // Also, some contacts can be destroyed.
00822                 m_contactManager.FindNewContacts();
00823         }
00824 
00825         m_stackAllocator.Free(queue);
