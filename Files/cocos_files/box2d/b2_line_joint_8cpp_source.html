00023 // Linear constraint (point-to-line)
00024 // d = p2 - p1 = x2 + r2 - x1 - r1
00025 // C = dot(perp, d)
00026 // Cdot = dot(d, cross(w1, perp)) + dot(perp, v2 + cross(w2, r2) - v1 - cross(w1, r1))
00027 //      = -dot(perp, v1) - dot(cross(d + r1, perp), w1) + dot(perp, v2) + dot(cross(r2, perp), v2)
00028 // J = [-perp, -cross(d + r1, perp), perp, cross(r2,perp)]
00029 //
00030 // K = J * invM * JT
00031 //
00032 // J = [-a -s1 a s2]
00033 // a = perp
00034 // s1 = cross(d + r1, a) = cross(p2 - x1, a)
00035 // s2 = cross(r2, a) = cross(p2 - x2, a)
00036 
00037 
00038 // Motor/Limit linear constraint
00039 // C = dot(ax1, d)
00040 // Cdot = = -dot(ax1, v1) - dot(cross(d + r1, ax1), w1) + dot(ax1, v2) + dot(cross(r2, ax1), v2)
00041 // J = [-ax1 -cross(d+r1,ax1) ax1 cross(r2,ax1)]
00042 
00043 // Block Solver
00044 // We develop a block solver that includes the joint limit. This makes the limit stiff (inelastic) even
00045 // when the mass has poor distribution (leading to large torques about the joint anchor points).
00046 //
00047 // The Jacobian has 3 rows:
00048 // J = [-uT -s1 uT s2] // linear
00049 //     [-vT -a1 vT a2] // limit
00050 //
00051 // u = perp
00052 // v = axis
00053 // s1 = cross(d + r1, u), s2 = cross(r2, u)
00054 // a1 = cross(d + r1, v), a2 = cross(r2, v)
00055 
00056 // M * (v2 - v1) = JT * df
00057 // J * v2 = bias
00058 //
00059 // v2 = v1 + invM * JT * df
00060 // J * (v1 + invM * JT * df) = bias
00061 // K * df = bias - J * v1 = -Cdot
00062 // K = J * invM * JT
00063 // Cdot = J * v1 - bias
00064 //
00065 // Now solve for f2.
00066 // df = f2 - f1
00067 // K * (f2 - f1) = -Cdot
00068 // f2 = invK * (-Cdot) + f1
00069 //
00070 // Clamp accumulated limit impulse.
00071 // lower: f2(2) = max(f2(2), 0)
00072 // upper: f2(2) = min(f2(2), 0)
00073 //
00074 // Solve for correct f2(1)
00075 // K(1,1) * f2(1) = -Cdot(1) - K(1,2) * f2(2) + K(1,1:2) * f1
00076 //                = -Cdot(1) - K(1,2) * f2(2) + K(1,1) * f1(1) + K(1,2) * f1(2)
00077 // K(1,1) * f2(1) = -Cdot(1) - K(1,2) * (f2(2) - f1(2)) + K(1,1) * f1(1)
00078 // f2(1) = invK(1,1) * (-Cdot(1) - K(1,2) * (f2(2) - f1(2))) + f1(1)
00079 //
00080 // Now compute impulse to be applied:
00081 // df = f2 - f1

00190                         if (m_limitState != e_atUpperLimit)

00198                         m_limitState = e_inactiveLimit;

00200                 }

00205         }
00206 
00207         if (m_enableMotor == false)

00209                 m_motorImpulse = 0.0f;
00210         }

00213         {
00214                 // Account for variable time step.

00218                 b2Vec2 P = m_impulse.x * m_perp + (m_motorImpulse + m_impulse.y) * m_axis;

00223                 b1->m_angularVelocity -= m_invI1 * L1;

00226                 b2->m_angularVelocity += m_invI2 * L2;
00227         }
00228         else
00229         {
00230                 m_impulse.SetZero();
00231                 m_motorImpulse = 0.0f;
00232         }

00235 void b2LineJoint::SolveVelocityConstraints(const b2TimeStep& step)
00236 {
00237         b2Body* b1 = m_bodyA;
00238         b2Body* b2 = m_bodyB;
00239 
00240         b2Vec2 v1 = b1->m_linearVelocity;
00241         float32 w1 = b1->m_angularVelocity;
00242         b2Vec2 v2 = b2->m_linearVelocity;
00243         float32 w2 = b2->m_angularVelocity;
00244 
00245         // Solve linear motor constraint.
00246         if (m_enableMotor && m_limitState != e_equalLimits)
00247         {
00248                 float32 Cdot = b2Dot(m_axis, v2 - v1) + m_a2 * w2 - m_a1 * w1;
00249                 float32 impulse = m_motorMass * (m_motorSpeed - Cdot);
00250                 float32 oldImpulse = m_motorImpulse;
00251                 float32 maxImpulse = step.dt * m_maxMotorForce;
00252                 m_motorImpulse = b2Clamp(m_motorImpulse + impulse, -maxImpulse, maxImpulse);
00253                 impulse = m_motorImpulse - oldImpulse;
00254 
00255                 b2Vec2 P = impulse * m_axis;
00256                 float32 L1 = impulse * m_a1;
00257                 float32 L2 = impulse * m_a2;
00258 
00259                 v1 -= m_invMass1 * P;
00260                 w1 -= m_invI1 * L1;
00261 
00262                 v2 += m_invMass2 * P;
00263                 w2 += m_invI2 * L2;
00264         }
00265 
00266         float32 Cdot1 = b2Dot(m_perp, v2 - v1) + m_s2 * w2 - m_s1 * w1;
00267 
00268         if (m_enableLimit && m_limitState != e_inactiveLimit)
00269         {
00270                 // Solve prismatic and limit constraint in block form.
00271                 float32 Cdot2 = b2Dot(m_axis, v2 - v1) + m_a2 * w2 - m_a1 * w1;
00272                 b2Vec2 Cdot(Cdot1, Cdot2);
00273 
00274                 b2Vec2 f1 = m_impulse;
00275                 b2Vec2 df =  m_K.Solve(-Cdot);
00276                 m_impulse += df;
00277 
00278                 if (m_limitState == e_atLowerLimit)
00279                 {
00280                         m_impulse.y = b2Max(m_impulse.y, 0.0f);
00281                 }
00282                 else if (m_limitState == e_atUpperLimit)
00283                 {
00284                         m_impulse.y = b2Min(m_impulse.y, 0.0f);
00285                 }
00286 
00287                 // f2(1) = invK(1,1) * (-Cdot(1) - K(1,2) * (f2(2) - f1(2))) + f1(1)
00288                 float32 b = -Cdot1 - (m_impulse.y - f1.y) * m_K.col2.x;
00289                 float32 f2r;
00290                 if (m_K.col1.x != 0.0f)
00291                 {
00292                         f2r = b / m_K.col1.x + f1.x;
00293                 }
00294                 else
00295                 {
00296                         f2r = f1.x;     
00297                 }
00298 
00299                 m_impulse.x = f2r;
00300 
00301                 df = m_impulse - f1;
00302 
00303                 b2Vec2 P = df.x * m_perp + df.y * m_axis;
00304                 float32 L1 = df.x * m_s1 + df.y * m_a1;
00305                 float32 L2 = df.x * m_s2 + df.y * m_a2;
00306 
00307                 v1 -= m_invMass1 * P;
00308                 w1 -= m_invI1 * L1;
00309 
00310                 v2 += m_invMass2 * P;
00311                 w2 += m_invI2 * L2;
00312         }
00313         else
00314         {
00315                 // Limit is inactive, just solve the prismatic constraint in block form.
00316                 float32 df;
00317                 if (m_K.col1.x != 0.0f)
00318                 {
00319                         df = - Cdot1 / m_K.col1.x;
00320                 }
00321                 else
00322                 {
00323                         df = 0.0f;
00324                 }
00325                 m_impulse.x += df;
00326 
00327                 b2Vec2 P = df * m_perp;
00328                 float32 L1 = df * m_s1;
00329                 float32 L2 = df * m_s2;
00330 
00331                 v1 -= m_invMass1 * P;
00332                 w1 -= m_invI1 * L1;
00333 
00334                 v2 += m_invMass2 * P;
00335                 w2 += m_invI2 * L2;
00336         }
00337 
00338         b1->m_linearVelocity = v1;
00339         b1->m_angularVelocity = w1;
00340         b2->m_linearVelocity = v2;
00341         b2->m_angularVelocity = w2;
00342 }
