00023 // Linear constraint (point-to-line)
00024 // d = p2 - p1 = x2 + r2 - x1 - r1
00025 // C = dot(perp, d)
00026 // Cdot = dot(d, cross(w1, perp)) + dot(perp, v2 + cross(w2, r2) - v1 - cross(w1, r1))
00027 //      = -dot(perp, v1) - dot(cross(d + r1, perp), w1) + dot(perp, v2) + dot(cross(r2, perp), v2)
00028 // J = [-perp, -cross(d + r1, perp), perp, cross(r2,perp)]
00029 //
00030 // Angular constraint
00031 // C = a2 - a1 + a_initial
00032 // Cdot = w2 - w1
00033 // J = [0 0 -1 0 0 1]
00034 //
00035 // K = J * invM * JT
00036 //
00037 // J = [-a -s1 a s2]
00038 //     [0  -1  0  1]
00039 // a = perp
00040 // s1 = cross(d + r1, a) = cross(p2 - x1, a)
00041 // s2 = cross(r2, a) = cross(p2 - x2, a)
00042 
00043 
00044 // Motor/Limit linear constraint
00045 // C = dot(ax1, d)
00046 // Cdot = = -dot(ax1, v1) - dot(cross(d + r1, ax1), w1) + dot(ax1, v2) + dot(cross(r2, ax1), v2)
00047 // J = [-ax1 -cross(d+r1,ax1) ax1 cross(r2,ax1)]
00048 
00049 // Block Solver
00050 // We develop a block solver that includes the joint limit. This makes the limit stiff (inelastic) even
00051 // when the mass has poor distribution (leading to large torques about the joint anchor points).
00052 //
00053 // The Jacobian has 3 rows:
00054 // J = [-uT -s1 uT s2] // linear
00055 //     [0   -1   0  1] // angular
00056 //     [-vT -a1 vT a2] // limit
00057 //
00058 // u = perp
00059 // v = axis
00060 // s1 = cross(d + r1, u), s2 = cross(r2, u)
00061 // a1 = cross(d + r1, v), a2 = cross(r2, v)
00062 
00063 // M * (v2 - v1) = JT * df
00064 // J * v2 = bias
00065 //
00066 // v2 = v1 + invM * JT * df
00067 // J * (v1 + invM * JT * df) = bias
00068 // K * df = bias - J * v1 = -Cdot
00069 // K = J * invM * JT
00070 // Cdot = J * v1 - bias
00071 //
00072 // Now solve for f2.
00073 // df = f2 - f1
00074 // K * (f2 - f1) = -Cdot
00075 // f2 = invK * (-Cdot) + f1
00076 //
00077 // Clamp accumulated limit impulse.
00078 // lower: f2(3) = max(f2(3), 0)
00079 // upper: f2(3) = min(f2(3), 0)
00080 //
00081 // Solve for correct f2(1:2)
00082 // K(1:2, 1:2) * f2(1:2) = -Cdot(1:2) - K(1:2,3) * f2(3) + K(1:2,1:3) * f1
00083 //                       = -Cdot(1:2) - K(1:2,3) * f2(3) + K(1:2,1:2) * f1(1:2) + K(1:2,3) * f1(3)
00084 // K(1:2, 1:2) * f2(1:2) = -Cdot(1:2) - K(1:2,3) * (f2(3) - f1(3)) + K(1:2,1:2) * f1(1:2)
00085 // f2(1:2) = invK(1:2,1:2) * (-Cdot(1:2) - K(1:2,3) * (f2(3) - f1(3))) + f1(1:2)
00086 //
00087 // Now compute impulse to be applied:
00088 // df = f2 - f1

00125 void b2PrismaticJoint::InitVelocityConstraints(const b2TimeStep& step)
00126 {
00127         b2Body* b1 = m_bodyA;
00128         b2Body* b2 = m_bodyB;
00129 
00130         // You cannot create a prismatic joint between bodies that
00131         // both have fixed rotation.
00132         b2Assert(b1->m_invI > 0.0f || b2->m_invI > 0.0f);
00133 
00134         m_localCenter1 = b1->GetLocalCenter();
00135         m_localCenter2 = b2->GetLocalCenter();
00136 
00137         b2Transform xf1 = b1->GetTransform();
00138         b2Transform xf2 = b2->GetTransform();
00139 
00140         // Compute the effective masses.
00141         b2Vec2 r1 = b2Mul(xf1.R, m_localAnchor1 - m_localCenter1);
00142         b2Vec2 r2 = b2Mul(xf2.R, m_localAnchor2 - m_localCenter2);
00143         b2Vec2 d = b2->m_sweep.c + r2 - b1->m_sweep.c - r1;
00144 
00145         m_invMass1 = b1->m_invMass;
00146         m_invI1 = b1->m_invI;
00147         m_invMass2 = b2->m_invMass;
00148         m_invI2 = b2->m_invI;
00149 
00150         // Compute motor Jacobian and effective mass.
00151         {
00152                 m_axis = b2Mul(xf1.R, m_localXAxis1);
00153                 m_a1 = b2Cross(d + r1, m_axis);
00154                 m_a2 = b2Cross(r2, m_axis);
00155 
00156                 m_motorMass = m_invMass1 + m_invMass2 + m_invI1 * m_a1 * m_a1 + m_invI2 * m_a2 * m_a2;
00157                 if (m_motorMass > B2_FLT_EPSILON)
00158                 {
00159                         m_motorMass = 1.0f / m_motorMass;
00160                 }
00161         }
00162 
00163         // Prismatic constraint.
00164         {
00165                 m_perp = b2Mul(xf1.R, m_localYAxis1);
00166 
00167                 m_s1 = b2Cross(d + r1, m_perp);
00168                 m_s2 = b2Cross(r2, m_perp);
00169 
00170                 float32 m1 = m_invMass1, m2 = m_invMass2;
00171                 float32 i1 = m_invI1, i2 = m_invI2;
00172 
00173                 float32 k11 = m1 + m2 + i1 * m_s1 * m_s1 + i2 * m_s2 * m_s2;
00174                 float32 k12 = i1 * m_s1 + i2 * m_s2;
00175                 float32 k13 = i1 * m_s1 * m_a1 + i2 * m_s2 * m_a2;
00176                 float32 k22 = i1 + i2;
00177                 float32 k23 = i1 * m_a1 + i2 * m_a2;
00178                 float32 k33 = m1 + m2 + i1 * m_a1 * m_a1 + i2 * m_a2 * m_a2;
00179 
00180                 m_K.col1.Set(k11, k12, k13);
00181                 m_K.col2.Set(k12, k22, k23);
00182                 m_K.col3.Set(k13, k23, k33);
00183         }
00184 
00185         // Compute motor and limit terms.
00186         if (m_enableLimit)
00187         {
00188                 float32 jointTranslation = b2Dot(m_axis, d);
00189                 if (b2Abs(m_upperTranslation - m_lowerTranslation) < 2.0f * b2_linearSlop)
00190                 {
00191                         m_limitState = e_equalLimits;
00192                 }
00193                 else if (jointTranslation <= m_lowerTranslation)
00194                 {
00195                         if (m_limitState != e_atLowerLimit)
00196                         {
00197                                 m_limitState = e_atLowerLimit;
00198                                 m_impulse.z = 0.0f;
00199                         }
00200                 }
00201                 else if (jointTranslation >= m_upperTranslation)
00202                 {
00203                         if (m_limitState != e_atUpperLimit)
00204                         {
00205                                 m_limitState = e_atUpperLimit;
00206                                 m_impulse.z = 0.0f;
00207                         }
00208                 }
00209                 else
00210                 {
00211                         m_limitState = e_inactiveLimit;
00212                         m_impulse.z = 0.0f;
00213                 }
00214         }
00215         else
00216         {
00217                 m_limitState = e_inactiveLimit;
00218         }
00219 
00220         if (m_enableMotor == false)
00221         {
00222                 m_motorImpulse = 0.0f;
00223         }
00224 
00225         if (step.warmStarting)
00226         {
00227                 // Account for variable time step.
00228                 m_impulse *= step.dtRatio;
00229                 m_motorImpulse *= step.dtRatio;
00230 
00231                 b2Vec2 P = m_impulse.x * m_perp + (m_motorImpulse + m_impulse.z) * m_axis;
00232                 float32 L1 = m_impulse.x * m_s1 + m_impulse.y + (m_motorImpulse + m_impulse.z) * m_a1;
00233                 float32 L2 = m_impulse.x * m_s2 + m_impulse.y + (m_motorImpulse + m_impulse.z) * m_a2;
00234 
00235                 b1->m_linearVelocity -= m_invMass1 * P;
00236                 b1->m_angularVelocity -= m_invI1 * L1;
00237 
00238                 b2->m_linearVelocity += m_invMass2 * P;
00239                 b2->m_angularVelocity += m_invI2 * L2;
00240         }
00241         else
00242         {
00243                 m_impulse.SetZero();
00244                 m_motorImpulse = 0.0f;
00245         }
00246 }

00248 void b2PrismaticJoint::SolveVelocityConstraints(const b2TimeStep& step)
00249 {
00250         b2Body* b1 = m_bodyA;
00251         b2Body* b2 = m_bodyB;
00252 
00253         b2Vec2 v1 = b1->m_linearVelocity;
00254         float32 w1 = b1->m_angularVelocity;
00255         b2Vec2 v2 = b2->m_linearVelocity;
00256         float32 w2 = b2->m_angularVelocity;
00257 
00258         // Solve linear motor constraint.
00259         if (m_enableMotor && m_limitState != e_equalLimits)
00260         {
00261                 float32 Cdot = b2Dot(m_axis, v2 - v1) + m_a2 * w2 - m_a1 * w1;
00262                 float32 impulse = m_motorMass * (m_motorSpeed - Cdot);
00263                 float32 oldImpulse = m_motorImpulse;
00264                 float32 maxImpulse = step.dt * m_maxMotorForce;
00265                 m_motorImpulse = b2Clamp(m_motorImpulse + impulse, -maxImpulse, maxImpulse);
00266                 impulse = m_motorImpulse - oldImpulse;
00267 
00268                 b2Vec2 P = impulse * m_axis;
00269                 float32 L1 = impulse * m_a1;
00270                 float32 L2 = impulse * m_a2;
00271 
00272                 v1 -= m_invMass1 * P;
00273                 w1 -= m_invI1 * L1;
00274 
00275                 v2 += m_invMass2 * P;
00276                 w2 += m_invI2 * L2;
00277         }
00278 
00279         b2Vec2 Cdot1;
00280         Cdot1.x = b2Dot(m_perp, v2 - v1) + m_s2 * w2 - m_s1 * w1;
00281         Cdot1.y = w2 - w1;
00282 
00283         if (m_enableLimit && m_limitState != e_inactiveLimit)
00284         {
00285                 // Solve prismatic and limit constraint in block form.
00286                 float32 Cdot2;
00287                 Cdot2 = b2Dot(m_axis, v2 - v1) + m_a2 * w2 - m_a1 * w1;
00288                 b2Vec3 Cdot(Cdot1.x, Cdot1.y, Cdot2);
00289 
00290                 b2Vec3 f1 = m_impulse;
00291                 b2Vec3 df =  m_K.Solve33(-Cdot);
00292                 m_impulse += df;
00293 
00294                 if (m_limitState == e_atLowerLimit)
00295                 {
00296                         m_impulse.z = b2Max(m_impulse.z, 0.0f);
00297                 }
00298                 else if (m_limitState == e_atUpperLimit)
00299                 {
00300                         m_impulse.z = b2Min(m_impulse.z, 0.0f);
00301                 }
00302 
00303                 // f2(1:2) = invK(1:2,1:2) * (-Cdot(1:2) - K(1:2,3) * (f2(3) - f1(3))) + f1(1:2)
00304                 b2Vec2 b = -Cdot1 - (m_impulse.z - f1.z) * b2Vec2(m_K.col3.x, m_K.col3.y);
00305                 b2Vec2 f2r = m_K.Solve22(b) + b2Vec2(f1.x, f1.y);
00306                 m_impulse.x = f2r.x;
00307                 m_impulse.y = f2r.y;
00308 
00309                 df = m_impulse - f1;
00310 
00311                 b2Vec2 P = df.x * m_perp + df.z * m_axis;
00312                 float32 L1 = df.x * m_s1 + df.y + df.z * m_a1;
00313                 float32 L2 = df.x * m_s2 + df.y + df.z * m_a2;
00314 
00315                 v1 -= m_invMass1 * P;
00316                 w1 -= m_invI1 * L1;
00317 
00318                 v2 += m_invMass2 * P;
00319                 w2 += m_invI2 * L2;
00320         }
00321         else
00322         {
00323                 // Limit is inactive, just solve the prismatic constraint in block form.
00324                 b2Vec2 df = m_K.Solve22(-Cdot1);
00325                 m_impulse.x += df.x;
00326                 m_impulse.y += df.y;
00327 
00328                 b2Vec2 P = df.x * m_perp;
00329                 float32 L1 = df.x * m_s1 + df.y;
00330                 float32 L2 = df.x * m_s2 + df.y;
00331 
00332                 v1 -= m_invMass1 * P;
00333                 w1 -= m_invI1 * L1;
00334 
00335                 v2 += m_invMass2 * P;
00336                 w2 += m_invI2 * L2;
00337         }
00338 
00339         b1->m_linearVelocity = v1;
00340         b1->m_angularVelocity = w1;
00341         b2->m_linearVelocity = v2;
00342         b2->m_angularVelocity = w2;
00343 }

00401
