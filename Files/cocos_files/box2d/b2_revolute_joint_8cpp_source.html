00023 // Point-to-point constraint
00024 // C = p2 - p1
00025 // Cdot = v2 - v1
00026 //      = v2 + cross(w2, r2) - v1 - cross(w1, r1)
00027 // J = [-I -r1_skew I r2_skew ]
00028 // Identity used:
00029 // w k % (rx i + ry j) = w * (-ry i + rx j)
00030 
00031 // Motor constraint
00032 // Cdot = w2 - w1
00033 // J = [0 0 -1 0 0 1]
00034 // K = invI1 + invI2

00064 void b2RevoluteJoint::InitVelocityConstraints(const b2TimeStep& step)
00065 {
00066         b2Body* b1 = m_bodyA;
00067         b2Body* b2 = m_bodyB;
00068 
00069         if (m_enableMotor || m_enableLimit)
00070         {
00071                 // You cannot create a rotation limit between bodies that
00072                 // both have fixed rotation.
00073                 b2Assert(b1->m_invI > 0.0f || b2->m_invI > 0.0f);
00074         }
00075 
00076         // Compute the effective mass matrix.
00077         b2Vec2 r1 = b2Mul(b1->GetTransform().R, m_localAnchor1 - b1->GetLocalCenter());
00078         b2Vec2 r2 = b2Mul(b2->GetTransform().R, m_localAnchor2 - b2->GetLocalCenter());
00079 
00080         // J = [-I -r1_skew I r2_skew]
00081         //     [ 0       -1 0       1]
00082         // r_skew = [-ry; rx]
00083 
00084         // Matlab
00085         // K = [ m1+r1y^2*i1+m2+r2y^2*i2,  -r1y*i1*r1x-r2y*i2*r2x,          -r1y*i1-r2y*i2]
00086         //     [  -r1y*i1*r1x-r2y*i2*r2x, m1+r1x^2*i1+m2+r2x^2*i2,           r1x*i1+r2x*i2]
00087         //     [          -r1y*i1-r2y*i2,           r1x*i1+r2x*i2,                   i1+i2]
00088 
00089         float32 m1 = b1->m_invMass, m2 = b2->m_invMass;
00090         float32 i1 = b1->m_invI, i2 = b2->m_invI;
00091 
00092         m_mass.col1.x = m1 + m2 + r1.y * r1.y * i1 + r2.y * r2.y * i2;
00093         m_mass.col2.x = -r1.y * r1.x * i1 - r2.y * r2.x * i2;
00094         m_mass.col3.x = -r1.y * i1 - r2.y * i2;
00095         m_mass.col1.y = m_mass.col2.x;
00096         m_mass.col2.y = m1 + m2 + r1.x * r1.x * i1 + r2.x * r2.x * i2;
00097         m_mass.col3.y = r1.x * i1 + r2.x * i2;
00098         m_mass.col1.z = m_mass.col3.x;
00099         m_mass.col2.z = m_mass.col3.y;
00100         m_mass.col3.z = i1 + i2;
00101 
00102         m_motorMass = 1.0f / (i1 + i2);
00103 
00104         if (m_enableMotor == false)
00105         {
00106                 m_motorImpulse = 0.0f;
00107         }
00108 
00109         if (m_enableLimit)
00110         {
00111                 float32 jointAngle = b2->m_sweep.a - b1->m_sweep.a - m_referenceAngle;
00112                 if (b2Abs(m_upperAngle - m_lowerAngle) < 2.0f * b2_angularSlop)
00113                 {
00114                         m_limitState = e_equalLimits;
00115                 }
00116                 else if (jointAngle <= m_lowerAngle)
00117                 {
00118                         if (m_limitState != e_atLowerLimit)
00119                         {
00120                                 m_impulse.z = 0.0f;
00121                         }
00122                         m_limitState = e_atLowerLimit;
00123                 }
00124                 else if (jointAngle >= m_upperAngle)
00125                 {
00126                         if (m_limitState != e_atUpperLimit)
00127                         {
00128                                 m_impulse.z = 0.0f;
00129                         }
00130                         m_limitState = e_atUpperLimit;
00131                 }
00132                 else
00133                 {
00134                         m_limitState = e_inactiveLimit;
00135                         m_impulse.z = 0.0f;
00136                 }
00137         }
00138         else
00139         {
00140                 m_limitState = e_inactiveLimit;
00141         }
00142 
00143         if (step.warmStarting)
00144         {
00145                 // Scale impulses to support a variable time step.
00146                 m_impulse *= step.dtRatio;
00147                 m_motorImpulse *= step.dtRatio;
00148 
00149                 b2Vec2 P(m_impulse.x, m_impulse.y);
00150 
00151                 b1->m_linearVelocity -= m1 * P;
00152                 b1->m_angularVelocity -= i1 * (b2Cross(r1, P) + m_motorImpulse + m_impulse.z);
00153 
00154                 b2->m_linearVelocity += m2 * P;
00155                 b2->m_angularVelocity += i2 * (b2Cross(r2, P) + m_motorImpulse + m_impulse.z);
00156         }
00157         else
00158         {
00159                 m_impulse.SetZero();
00160                 m_motorImpulse = 0.0f;
00161         }
00162 }

00164 void b2RevoluteJoint::SolveVelocityConstraints(const b2TimeStep& step)
00165 {
00166         b2Body* b1 = m_bodyA;
00167         b2Body* b2 = m_bodyB;
00168 
00169         b2Vec2 v1 = b1->m_linearVelocity;
00170         float32 w1 = b1->m_angularVelocity;
00171         b2Vec2 v2 = b2->m_linearVelocity;
00172         float32 w2 = b2->m_angularVelocity;
00173 
00174         float32 m1 = b1->m_invMass, m2 = b2->m_invMass;
00175         float32 i1 = b1->m_invI, i2 = b2->m_invI;
00176 
00177         // Solve motor constraint.
00178         if (m_enableMotor && m_limitState != e_equalLimits)
00179         {
00180                 float32 Cdot = w2 - w1 - m_motorSpeed;
00181                 float32 impulse = m_motorMass * (-Cdot);
00182                 float32 oldImpulse = m_motorImpulse;
00183                 float32 maxImpulse = step.dt * m_maxMotorTorque;
00184                 m_motorImpulse = b2Clamp(m_motorImpulse + impulse, -maxImpulse, maxImpulse);
00185                 impulse = m_motorImpulse - oldImpulse;
00186 
00187                 w1 -= i1 * impulse;
00188                 w2 += i2 * impulse;
00189         }
00190 
00191         // Solve limit constraint.
00192         if (m_enableLimit && m_limitState != e_inactiveLimit)
00193         {
00194                 b2Vec2 r1 = b2Mul(b1->GetTransform().R, m_localAnchor1 - b1->GetLocalCenter());
00195                 b2Vec2 r2 = b2Mul(b2->GetTransform().R, m_localAnchor2 - b2->GetLocalCenter());
00196 
00197                 // Solve point-to-point constraint
00198                 b2Vec2 Cdot1 = v2 + b2Cross(w2, r2) - v1 - b2Cross(w1, r1);
00199                 float32 Cdot2 = w2 - w1;
00200                 b2Vec3 Cdot(Cdot1.x, Cdot1.y, Cdot2);
00201 
00202                 b2Vec3 impulse = m_mass.Solve33(-Cdot);
00203 
00204                 if (m_limitState == e_equalLimits)
00205                 {
00206                         m_impulse += impulse;
00207                 }
00208                 else if (m_limitState == e_atLowerLimit)
00209                 {
00210                         float32 newImpulse = m_impulse.z + impulse.z;
00211                         if (newImpulse < 0.0f)
00212                         {
00213                                 b2Vec2 reduced = m_mass.Solve22(-Cdot1);
00214                                 impulse.x = reduced.x;
00215                                 impulse.y = reduced.y;
00216                                 impulse.z = -m_impulse.z;
00217                                 m_impulse.x += reduced.x;
00218                                 m_impulse.y += reduced.y;
00219                                 m_impulse.z = 0.0f;
00220                         }
00221                 }
00222                 else if (m_limitState == e_atUpperLimit)
00223                 {
00224                         float32 newImpulse = m_impulse.z + impulse.z;
00225                         if (newImpulse > 0.0f)
00226                         {
00227                                 b2Vec2 reduced = m_mass.Solve22(-Cdot1);
00228                                 impulse.x = reduced.x;
00229                                 impulse.y = reduced.y;
00230                                 impulse.z = -m_impulse.z;
00231                                 m_impulse.x += reduced.x;
00232                                 m_impulse.y += reduced.y;
00233                                 m_impulse.z = 0.0f;
00234                         }
00235                 }
00236 
00237                 b2Vec2 P(impulse.x, impulse.y);
00238 
00239                 v1 -= m1 * P;
00240                 w1 -= i1 * (b2Cross(r1, P) + impulse.z);
00241 
00242                 v2 += m2 * P;
00243                 w2 += i2 * (b2Cross(r2, P) + impulse.z);
00244         }
00245         else
00246         {
00247                 b2Vec2 r1 = b2Mul(b1->GetTransform().R, m_localAnchor1 - b1->GetLocalCenter());
00248                 b2Vec2 r2 = b2Mul(b2->GetTransform().R, m_localAnchor2 - b2->GetLocalCenter());
00249 
00250                 // Solve point-to-point constraint
00251                 b2Vec2 Cdot = v2 + b2Cross(w2, r2) - v1 - b2Cross(w1, r1);
00252                 b2Vec2 impulse = m_mass.Solve22(-Cdot);
00253 
00254                 m_impulse.x += impulse.x;
00255                 m_impulse.y += impulse.y;
00256 
00257                 v1 -= m1 * impulse;
00258                 w1 -= i1 * b2Cross(r1, impulse);
00259 
00260                 v2 += m2 * impulse;
00261                 w2 += i2 * b2Cross(r2, impulse);
00262         }
00263 
00264         b1->m_linearVelocity = v1;
00265         b1->m_angularVelocity = w1;
00266         b2->m_linearVelocity = v2;
00267         b2->m_angularVelocity = w2;

00270 bool b2RevoluteJoint::SolvePositionConstraints(float32 baumgarte)
00271 {
00272         // TODO_ERIN block solve with limit.
00273 
00274         B2_NOT_USED(baumgarte);
00275 
00276         b2Body* b1 = m_bodyA;
00277         b2Body* b2 = m_bodyB;
00278 
00279         float32 angularError = 0.0f;
00280         float32 positionError = 0.0f;
00281 
00282         // Solve angular limit constraint.
00283         if (m_enableLimit && m_limitState != e_inactiveLimit)
00284         {
00285                 float32 angle = b2->m_sweep.a - b1->m_sweep.a - m_referenceAngle;
00286                 float32 limitImpulse = 0.0f;
00287 
00288                 if (m_limitState == e_equalLimits)
00289                 {
00290                         // Prevent large angular corrections
00291                         float32 C = b2Clamp(angle - m_lowerAngle, -b2_maxAngularCorrection, b2_maxAngularCorrection);
00292                         limitImpulse = -m_motorMass * C;
00293                         angularError = b2Abs(C);
00294                 }
00295                 else if (m_limitState == e_atLowerLimit)
00296                 {
00297                         float32 C = angle - m_lowerAngle;
00298                         angularError = -C;
00299 
00300                         // Prevent large angular corrections and allow some slop.
00301                         C = b2Clamp(C + b2_angularSlop, -b2_maxAngularCorrection, 0.0f);
00302                         limitImpulse = -m_motorMass * C;
00303                 }
00304                 else if (m_limitState == e_atUpperLimit)
00305                 {
00306                         float32 C = angle - m_upperAngle;
00307                         angularError = C;
00308 
00309                         // Prevent large angular corrections and allow some slop.
00310                         C = b2Clamp(C - b2_angularSlop, 0.0f, b2_maxAngularCorrection);
00311                         limitImpulse = -m_motorMass * C;
00312                 }
00313 
00314                 b1->m_sweep.a -= b1->m_invI * limitImpulse;
00315                 b2->m_sweep.a += b2->m_invI * limitImpulse;
00316 
00317                 b1->SynchronizeTransform();
00318                 b2->SynchronizeTransform();
00319         }
00320 
00321         // Solve point-to-point constraint.
00322         {
00323                 b2Vec2 r1 = b2Mul(b1->GetTransform().R, m_localAnchor1 - b1->GetLocalCenter());
00324                 b2Vec2 r2 = b2Mul(b2->GetTransform().R, m_localAnchor2 - b2->GetLocalCenter());
00325 
00326                 b2Vec2 C = b2->m_sweep.c + r2 - b1->m_sweep.c - r1;
00327                 positionError = C.Length();
00328 
00329                 float32 invMass1 = b1->m_invMass, invMass2 = b2->m_invMass;
00330                 float32 invI1 = b1->m_invI, invI2 = b2->m_invI;
00331 
00332                 // Handle large detachment.
00333                 const float32 k_allowedStretch = 10.0f * b2_linearSlop;
00334                 if (C.LengthSquared() > k_allowedStretch * k_allowedStretch)
00335                 {
00336                         // Use a particle solution (no rotation).
00337                         b2Vec2 u = C; u.Normalize();
00338                         float32 k = invMass1 + invMass2;
00339                         b2Assert(k > B2_FLT_EPSILON);
00340                         float32 m = 1.0f / k;
00341                         b2Vec2 impulse = m * (-C);
00342                         const float32 k_beta = 0.5f;
00343                         b1->m_sweep.c -= k_beta * invMass1 * impulse;
00344                         b2->m_sweep.c += k_beta * invMass2 * impulse;
00345 
00346                         C = b2->m_sweep.c + r2 - b1->m_sweep.c - r1;
00347                 }
00348 
00349                 b2Mat22 K1;
00350                 K1.col1.x = invMass1 + invMass2;        K1.col2.x = 0.0f;
00351                 K1.col1.y = 0.0f;                                       K1.col2.y = invMass1 + invMass2;
00352 
00353                 b2Mat22 K2;
00354                 K2.col1.x =  invI1 * r1.y * r1.y;       K2.col2.x = -invI1 * r1.x * r1.y;
00355                 K2.col1.y = -invI1 * r1.x * r1.y;       K2.col2.y =  invI1 * r1.x * r1.x;
00356 
00357                 b2Mat22 K3;
00358                 K3.col1.x =  invI2 * r2.y * r2.y;       K3.col2.x = -invI2 * r2.x * r2.y;
00359                 K3.col1.y = -invI2 * r2.x * r2.y;       K3.col2.y =  invI2 * r2.x * r2.x;
00360 
00361                 b2Mat22 K = K1 + K2 + K3;
00362                 b2Vec2 impulse = K.Solve(-C);
00363 
00364                 b1->m_sweep.c -= b1->m_invMass * impulse;
00365                 b1->m_sweep.a -= b1->m_invI * b2Cross(r1, impulse);
00366 
00367                 b2->m_sweep.c += b2->m_invMass * impulse;
00368                 b2->m_sweep.a += b2->m_invI * b2Cross(r2, impulse);
00369 
00370                 b1->SynchronizeTransform();
00371                 b2->SynchronizeTransform();
00372         }
00373         
00374         return positionError <= b2_linearSlop && angularError <= b2_angularSlop;
