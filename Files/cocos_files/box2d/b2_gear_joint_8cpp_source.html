00025 // Gear Joint:
00026 // C0 = (coordinate1 + ratio * coordinate2)_initial
00027 // C = C0 - (cordinate1 + ratio * coordinate2) = 0
00028 // Cdot = -(Cdot1 + ratio * Cdot2)
00029 // J = -[J1 ratio * J2]
00030 // K = J * invM * JT
00031 //   = J1 * invM1 * J1T + ratio * ratio * J2 * invM2 * J2T
00032 //
00033 // Revolute:
00034 // coordinate = rotation
00035 // Cdot = angularVelocity
00036 // J = [0 0 1]
00037 // K = J * invM * JT = invI
00038 //
00039 // Prismatic:
00040 // coordinate = dot(p - pg, ug)
00041 // Cdot = dot(v + cross(w, r), ug)
00042 // J = [ug cross(r, ug)]
00043 // K = J * invM * JT = invMass + invI * cross(r, ug)^2

00104 void b2GearJoint::InitVelocityConstraints(const b2TimeStep& step)
00105 {
00106         b2Body* g1 = m_ground1;
00107         b2Body* g2 = m_ground2;
00108         b2Body* b1 = m_bodyA;
00109         b2Body* b2 = m_bodyB;

00111         float32 K = 0.0f;
00112         m_J.SetZero();
00113 
00114         if (m_revolute1)
00115         {
00116                 m_J.angular1 = -1.0f;

00119         else
00120         {
00121                 b2Vec2 ug = b2Mul(g1->GetTransform().R, m_prismatic1->m_localXAxis1);
00122                 b2Vec2 r = b2Mul(b1->GetTransform().R, m_localAnchor1 - b1->GetLocalCenter());

00124                 m_J.linear1 = -ug;
00125                 m_J.angular1 = -crug;
00126                 K += b1->m_invMass + b1->m_invI * crug * crug;
00127         }
00128 
00129         if (m_revolute2)
00130         {
00131                 m_J.angular2 = -m_ratio;
00132                 K += m_ratio * m_ratio * b2->m_invI;
00133         }
00134         else
00135         {
00136                 b2Vec2 ug = b2Mul(g2->GetTransform().R, m_prismatic2->m_localXAxis1);
00137                 b2Vec2 r = b2Mul(b2->GetTransform().R, m_localAnchor2 - b2->GetLocalCenter());
00138                 float32 crug = b2Cross(r, ug);
00139                 m_J.linear2 = -m_ratio * ug;
00140                 m_J.angular2 = -m_ratio * crug;
00141                 K += m_ratio * m_ratio * (b2->m_invMass + b2->m_invI * crug * crug);
00142         }
00143 
00144         // Compute effective mass.
00145         m_mass = K > 0.0f ? 1.0f / K : 0.0f;
00146 
00147         if (step.warmStarting)
00148         {
00149                 // Warm starting.
00150                 b1->m_linearVelocity += b1->m_invMass * m_impulse * m_J.linear1;
00151                 b1->m_angularVelocity += b1->m_invI * m_impulse * m_J.angular1;

00153                 b2->m_angularVelocity += b2->m_invI * m_impulse * m_J.angular2;
